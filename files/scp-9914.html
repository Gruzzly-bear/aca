import pyscp
import aiad

#Item

scp9914 = pyscp.scp(
item = 9914,
object_class = 'keter',
disruption_class = 'ekhi',
risk_class = 'danger')

#Special Containment Procedures

def labyrinth(mazerunner, mazemaker):

maze = mazemaker.createMaze()
maze.setRunner(mazerunner)

while maze.hasRunner == True:
if maze.getRunnerPosition() == maze.getExit():
maze = mazemaker.shiftMaze(maze)

#Description

prisioner = aiad.load("minotaur.aiad")

architect= aiad.load("wiseman.aiad")

scp9914.contain(
containment_function = labyrinth,
mazerunner = prisioner,
mazemaker = architect)
